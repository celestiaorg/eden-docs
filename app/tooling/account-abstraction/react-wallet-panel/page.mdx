## Usage

```tsx copy
import { WalletPanel } from '@wallet-panel/react'

function App() {
  return <WalletPanel privyClient={privyClient} onRequestLogin={() => openPrivyLogin()} />
}
```

## Usage with ZeroDev (smart accounts)

```tsx copy
import { WalletPanel } from '@wallet-panel/react'

function App() {
  return (
    <WalletPanel
      privyClient={privyClient}
      zerodev={zerodevContext}
      enableSponsoredTx={true}
      onRequestLogin={() => openPrivyLogin()}
      onTxSubmitted={hash => console.log('Transaction:', hash)}
    />
  )
}
```
